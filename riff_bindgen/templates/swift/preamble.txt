private enum StreamPollResult: Int8 {
    case ready = 0
    case closed = 1
}

@inline(__always)
private func stringFromFfi(_ ffiString: FfiString) -> String {
    guard ffiString.len > 0, let pointer = ffiString.ptr else { return "" }
    return String(decoding: UnsafeBufferPointer(start: pointer, count: Int(ffiString.len)), as: UTF8.self)
}

@inline(__always)
private func checkStatus(_ status: FfiStatus, context: StaticString = #function) throws {
    guard status.code == 0 else {
        throw FfiError(message: "FFI failed in \(context) with code \(status.code)")
    }
}

@inline(__always)
private func ensureOk(_ status: FfiStatus, context: StaticString = #function) {
    guard status.code == 0 else {
        fatalError("FFI failed in \(context) [\(status.code)]")
    }
}

    typealias Continuation = CheckedContinuation<T, Error>

    enum FinishDecision {
        case alreadyFinished
        case finishWithoutContinuation
        case finishWithContinuation(Continuation)
    }

    final class ContinuationBox {
        let continuation: Continuation
        init(_ continuation: Continuation) { self.continuation = continuation }
    }

    let handle: RustFutureHandle?
    private var continuationSlot: UInt64 = 0

    init(handle: RustFutureHandle?) {
        self.handle = handle
    }

    func installContinuation(_ continuation: Continuation) -> Bool {
        let box = ContinuationBox(continuation)
        let raw = UInt64(UInt(bitPattern: Unmanaged.passRetained(box).toOpaque()))
        let prior = withUnsafeMutablePointer(to: &continuationSlot) { {{ prefix }}_atomic_u64_exchange($0, raw) }

        if prior == 0 { return true }
        if prior == 1 {
            Unmanaged.passUnretained(box).release()
            return false
        }
        withUnsafeMutablePointer(to: &continuationSlot) { _ = {{ prefix }}_atomic_u64_exchange($0, prior) }
        Unmanaged.passUnretained(box).release()
        return false
    }

    @inline(__always)
    func canPoll() -> Bool {
        withUnsafeMutablePointer(to: &continuationSlot) { {{ prefix }}_atomic_u64_load($0) } != 1
    }

    func decideFinish() -> FinishDecision {
        let prior = withUnsafeMutablePointer(to: &continuationSlot) { {{ prefix }}_atomic_u64_exchange($0, 1) }
        if prior == 1 { return .alreadyFinished }
        if prior == 0 { return .finishWithoutContinuation }
        let box = Unmanaged<ContinuationBox>.fromOpaque(UnsafeRawPointer(bitPattern: UInt(prior))!).takeRetainedValue()
        return .finishWithContinuation(box.continuation)
    }

struct WireBuffer {
    let data: Data
    init(ptr: UnsafePointer<UInt8>, len: UInt) { self.data = Data(bytes: ptr, count: Int(len)) }
    init(data: Data) { self.data = data }
    @inlinable var contentOffset: Int { 10 }
    @inlinable func readU8(at offset: Int) -> UInt8 { data[offset] }
    @inlinable func readI8(at offset: Int) -> Int8 { Int8(bitPattern: data[offset]) }
    @inlinable func readU16(at offset: Int) -> UInt16 { data.withUnsafeBytes { $0.loadUnaligned(fromByteOffset: offset, as: UInt16.self) } }
    @inlinable func readI16(at offset: Int) -> Int16 { Int16(bitPattern: readU16(at: offset)) }
    @inlinable func readU32(at offset: Int) -> UInt32 { data.withUnsafeBytes { $0.loadUnaligned(fromByteOffset: offset, as: UInt32.self) } }
    @inlinable func readI32(at offset: Int) -> Int32 { Int32(bitPattern: readU32(at: offset)) }
    @inlinable func readU64(at offset: Int) -> UInt64 { data.withUnsafeBytes { $0.loadUnaligned(fromByteOffset: offset, as: UInt64.self) } }
    @inlinable func readI64(at offset: Int) -> Int64 { Int64(bitPattern: readU64(at: offset)) }
    @inlinable func readF32(at offset: Int) -> Float { Float(bitPattern: readU32(at: offset)) }
    @inlinable func readF64(at offset: Int) -> Double { Double(bitPattern: readU64(at: offset)) }
    @inlinable func readBool(at offset: Int) -> Bool { data[offset] != 0 }
    @inlinable func readString(at offset: Int) -> (value: String, size: Int) {
        let len = Int(readU32(at: offset))
        guard len > 0 else { return ("", 4) }
        let start = offset + 4
        return (String(decoding: data[start..<(start + len)], as: UTF8.self), 4 + len)
    }
    @inlinable func readOptional<T>(at offset: Int, reader: (Int) -> (value: T, size: Int)) -> (value: T?, size: Int) {
        guard data[offset] != 0 else { return (nil, 1) }
        let (value, size) = reader(offset + 1)
        return (value, 1 + size)
    }
    @inlinable func readFixedArray<T>(at offset: Int, elementSize: Int, reader: (Int) -> T) -> (value: [T], size: Int) {
        let count = Int(readU32(at: offset))
        guard count > 0 else { return ([], 4) }
        var result = [T]()
        result.reserveCapacity(count)
        var pos = offset + 4
        for _ in 0..<count { result.append(reader(pos)); pos += elementSize }
        return (result, pos - offset)
    }
    @inlinable func readVariableArray<T>(at offset: Int, reader: (Int) -> (value: T, size: Int)) -> (value: [T], size: Int) {
        let count = Int(readU32(at: offset))
        guard count > 0 else { return ([], 4) }
        let offsetsStart = offset + 4
        var result = [T]()
        result.reserveCapacity(count)
        var maxEnd = offsetsStart + (count * 4)
        for i in 0..<count {
            let relOffset = Int(readU32(at: offsetsStart + (i * 4)))
            let elemOffset = offsetsStart + relOffset
            let (value, elemSize) = reader(elemOffset)
            result.append(value)
            let elemEnd = elemOffset + elemSize
            if elemEnd > maxEnd { maxEnd = elemEnd }
        }
        return (result, maxEnd - offset)
    }
}
